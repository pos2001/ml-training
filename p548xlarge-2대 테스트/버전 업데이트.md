

## 컴퓨트 노드 EFA 업그레이드 스크립트
```
cat > /fsx/force-upgrade-efa.sh << 'EOF'
#!/bin/bash
set -e

echo "=== Force upgrading EFA to 1.26.1 on $(hostname) ==="

# 현재 버전
echo "Current version:"
/opt/amazon/efa/bin/fi_info --version 2>/dev/null | head -1

cd /tmp

# 기존 제거
echo "Removing old EFA installation..."
if [ -d "/tmp/aws-efa-installer" ]; then
    cd /tmp/aws-efa-installer
    ./efa_installer.sh -y --uninstall 2>/dev/null || true
    cd /tmp
fi
rm -rf aws-efa-installer*

# 1.26.1 다운로드
echo "Downloading EFA 1.26.1..."
wget -q https://efa-installer.amazonaws.com/aws-efa-installer-1.26.1.tar.gz
tar -xf aws-efa-installer-1.26.1.tar.gz
cd aws-efa-installer

# 설치
echo "Installing EFA 1.26.1..."
./efa_installer.sh -y --skip-kmod --skip-limit-conf

# 확인
echo ""
echo "New version:"
/opt/amazon/efa/bin/fi_info --version 2>/dev/null | head -3

echo "✅ EFA upgrade complete on $(hostname)"
EOF

chmod +x /fsx/force-upgrade-efa.sh

# 실행
echo "=== Force Upgrading Compute Nodes to EFA 1.26.1 ==="
srun -N2 sudo bash /fsx/force-upgrade-efa.sh

```
